198  template <typename... ReturnType, typename... Args>
199  [[nodiscard]]
200  Detail::ScalarOrTupleT<ReturnType...> get(std::string_view sql, Args&&... args) {
201  auto row = exec(sql, std::forward<Args>(args)...);
202  if constexpr(Detail::IsScalarV<ReturnType...>) {
203  // Return a single value
204  if (row.columnCount() != 1)
205  throw Dbpp::Error(std::string("get() expects a single column Result. Statement: ") + std::string{ sql });
206  return row.template get<Detail::FirstTypeT<ReturnType...>>(0);
207  } else {
208  // Return it as a tuple
209  return row.template toTuple<ReturnType...>();
210  }
211  }
