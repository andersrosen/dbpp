159  template <typename... ReturnType, typename... Args>
160  [[nodiscard]]
161  Detail::ScalarOrTupleT<ReturnType...> get(std::string_view sql, Args... args) {
162  auto row = exec(sql, args...);
163  if constexpr(Detail::IsScalarV<ReturnType...>) {
164  // Return a single value
165  if (row.columnCount() != 1)
166  throw Dbpp::Error(std::string("get() expects a single column Result. Statement: ") + std::string{ sql });
167  return row.template get<Detail::FirstTypeT<ReturnType...>>(0);
168  } else {
169  // Return it as a tuple
170  return row.template toTuple<ReturnType...>();
171  }
