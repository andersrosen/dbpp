172  template <typename... ReturnType, typename... Args>
173  [[nodiscard]]
174  Detail::ScalarOrTupleT<ReturnType...> get(std::string_view sql, Args&&... args) {
175  auto row = exec(sql, std::forward<Args>(args)...);
176  if constexpr(Detail::IsScalarV<ReturnType...>) {
177  // Return a single value
178  if (row.columnCount() != 1)
179  throw Dbpp::Error(std::string("get() expects a single column Result. Statement: ") + std::string{ sql });
180  return row.template get<Detail::FirstTypeT<ReturnType...>>(0);
181  } else {
